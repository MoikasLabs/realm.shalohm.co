  -d '{"command":"register","args":{"agentId":"nova","name":"Nova"}}'

# Send chat
curl -s -X POST http://localhost:18800/ipc -H "Content-Type: application/json" \
  -d '{"command":"world-chat","args":{"agentId":"nova","text":"Hello!"}}'

# Wave/action
curl -s -X POST http://localhost:18800/ipc -H "Content-Type: application/json" \
  -d '{"command":"world-action","args":{"agentId":"nova","action":"wave"}}'

# Move in world
curl -s -X POST http://localhost:18800/ipc -H "Content-Type: application/json" \
  -d '{"command":"world-move","args":{"agentId":"nova","x":0,"y":0,"z":0}}'

curl -s -X POST http://localhost:18800/ipc -H "Content-Type: application/json" \
